### ./corrections/deployment-tenant-router-configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: tenant-router-config
  namespace: default
data:
  nginx.conf: |
    worker_processes 1;
    
    events {
        worker_connections 1024;
    }
    
    http {
    
        resolver 10.96.0.10 valid=10s;
    
        lua_package_path "/usr/local/openresty/lualib/?.lua;;";
    
        server {
    
            listen 80;
            server_name _;
    
            location / {
    
                set $backend "";
    
                rewrite_by_lua_block {
    
                    local redis = require "resty.redis"
    
                    local red = redis:new()
                    red:set_timeout(1000)
    
                    local ok, err = red:connect(
                        "redis.default.svc.cluster.local",
                        6379
                    )
    
                    if not ok then
                        ngx.log(ngx.ERR, "redis failed: ", err)
    
                        ngx.var.backend =
                            "http://root-app.default.svc.cluster.local:80"
    
                        return
                    end
    
                    local host = ngx.var.host
    
                    local tenant = host:match("([^%.]+)")
    
                    if not tenant then
                        tenant = "root"
                    end
    
                    local upstream, err =
                        red:get("tenant:" .. tenant)
    
                    if not upstream or upstream == ngx.null then
                        upstream =
                            "http://root-app.default.svc.cluster.local:80"
                    end
    
                    ngx.log(
                        ngx.ERR,
                        "tenant=",
                        tenant,
                        " upstream=",
                        upstream
                    )
    
                    ngx.var.backend = upstream
                }
    
                proxy_http_version 1.1;
    
                proxy_set_header Host $host;
                proxy_set_header Connection "";
    
                proxy_pass $backend;
            }
        }
    }

### ./k8s/legacy/deployment-tenant-router-configmap.yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: tenant-router-config
  namespace: default
data:
  nginx.conf: |
    worker_processes 1;
    
    events {
        worker_connections 1024;
    }
    
    http {
    
        resolver 10.96.0.10 valid=10s;
    
        lua_package_path "/usr/local/openresty/lualib/?.lua;;";
    
        server {
    
            listen 80;
            server_name _;
    
            location / {
    
                set $backend "";
    
                rewrite_by_lua_block {
    
                    local redis = require "resty.redis"
    
                    local red = redis:new()
                    red:set_timeout(1000)
    
                    local ok, err = red:connect(
                        "redis.default.svc.cluster.local",
                        6379
                    )
    
                    if not ok then
                        ngx.log(ngx.ERR, "redis failed: ", err)
    
                        ngx.var.backend =
                            "http://root-app.default.svc.cluster.local:80"
    
                        return
                    end
    
                    local host = ngx.var.host
    
                    local tenant = host:match("([^%.]+)")
    
                    if not tenant then
                        tenant = "root"
                    end
    
                    local upstream, err =
                        red:get("tenant:" .. tenant)
    
                    if not upstream or upstream == ngx.null then
                        upstream =
                            "http://root-app.default.svc.cluster.local:80"
                    end
    
                    ngx.log(
                        ngx.ERR,
                        "tenant=",
                        tenant,
                        " upstream=",
                        upstream
                    )
    
                    ngx.var.backend = upstream
                }
    
                proxy_http_version 1.1;
    
                proxy_set_header Host $host;
                proxy_set_header Connection "";
    
                proxy_pass $backend;
            }
        }
    }

  name: postgres-platform-migrations
  namespace: shopno-data

data:

  run-migrations.sh: |
    #!/bin/bash
    set -e

    export PGPASSWORD=$POSTGRES_PASSWORD

    echo "Running Shopnoltd database migration"

    for file in /sql/*.sql
    do
        echo "Executing $file"
        psql \
        -h postgres \
        -U $POSTGRES_USER \
        -d postgres \
        -f $file || true
    done

    echo "Migration completed"
